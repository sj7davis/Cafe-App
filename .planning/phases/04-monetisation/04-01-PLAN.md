```typescript
// ── Gift Cards ──────────────────────────────────────────────────────────────

createGiftCard:
  input: { token: string, amount: number(positive), senderName?: string,
           recipientName?: string, recipientPhone?: string, message?: string }
  auth: jwtVerify → venueId from payload
  action: generate 12-char code, INSERT giftCards with balance = amount
  returns: { id: number, code: string }

listGiftCards:
  input: { token: string }
  auth: jwtVerify → venueId from payload
  action: SELECT all giftCards for venueId ORDER BY createdAt DESC
  returns: array of giftCard rows

redeemGiftCard:
  input: { venueId: number(positive int), code: string(min 1), orderTotal: number(positive) }
  auth: none (public — called at checkout)
  action:
    1. SELECT giftCard WHERE venueId AND code.toUpperCase()
    2. throw NOT_FOUND if missing
    3. throw BAD_REQUEST if balance <= 0
    4. discount = min(balance, orderTotal)
    5. newBalance = balance - discount
    6. UPDATE balance = newBalance, isRedeemed = (newBalance <= 0)
  returns: { discount: number, remainingBalance: number }

// ── Subscription Passes ──────────────────────────────────────────────────────

upsertPassConfig:
  input: { token: string, name: string(min 1), totalCredits: number(int positive),
           price: number(positive) }
  auth: jwtVerify → venueId from payload
  action: UPDATE venues SET settingsJson = merge existing json with { passConfig: { name, totalCredits, price } }
  returns: { success: true }

getPassConfig:
  input: { venueId: number(positive int) }
  auth: none (public)
  action: SELECT venue by id, return (settingsJson as any)?.passConfig ?? null
  returns: { name: string, totalCredits: number, price: number } | null

purchasePass:
  input: { token: string, phone: string(min 1), name: string(min 1) }
  auth: jwtVerify → venueId from payload
  action:
    1. Read passConfig from venue settingsJson — throw BAD_REQUEST if not configured
    2. INSERT subscriptionPasses with totalCredits = remainingCredits = passConfig.totalCredits, price = passConfig.price
  returns: { id: number, remainingCredits: number }

getPassByPhone:
  input: { venueId: number(positive int), phone: string(min 1) }
  auth: none (public)
  action: SELECT subscriptionPasses WHERE venueId AND phone AND isActive=true
          ORDER BY createdAt DESC LIMIT 1
  returns: subscriptionPass row | null

usePassCredit:
  input: { passId: number(positive int), venueId: number(positive int) }
  auth: none (public — called from createOrder.onSuccess)
  action:
    1. SELECT pass WHERE id AND venueId AND isActive=true — throw NOT_FOUND if missing
    2. throw BAD_REQUEST if remainingCredits <= 0
    3. UPDATE remainingCredits = sql`remaining_credits - 1`
    4. if new remainingCredits reaches 0, also set isActive = false
  returns: { remainingCredits: number }
```
