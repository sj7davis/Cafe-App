<!-- Week navigation pattern -->
weekStart computed as Monday of current week:
  function getMonday(date: Date): string {
    const d = new Date(date)
    const day = d.getDay()
    const diff = d.getDate() - day + (day === 0 ? -6 : 1)
    d.setDate(diff)
    return d.toISOString().slice(0, 10)
  }
Navigation: prev/next week buttons adjust weekStart by ±7 days.
Display: "Week of 26 May 2026" format.

  const rows = [['Staff Name', 'Total Hours', 'Shifts', 'Penalty Flags']]
  for (const s of summary) rows.push([s.name, s.totalHours, String(s.shifts), s.penaltyFlags.join('; ')])
  const csv = rows.map(r => r.map(c => '"' + c.replace(/"/g, '""') + '"').join(',')).join('\n')
  const blob = new Blob([csv], { type: 'text/csv' })
  const url = URL.createObjectURL(blob)
  const a = document.createElement('a')
  a.href = url; a.download = 'timesheet.csv'; a.click()
